<script>
    document.addEventListener('DOMContentLoaded', () => {
        document.querySelectorAll('.read-more-button').forEach(button => {
            // Initialize ARIA attributes
            button.setAttribute('aria-expanded', 'false');
            button.setAttribute('aria-label', 'Expand bio content');
            button.setAttribute('role', 'button');
            
            // Initialize collapsed content attributes
            const parent = button.parentElement;
            const bioContentText = parent.querySelector('.bio-content-text');
            if (bioContentText && bioContentText.classList.contains('collapsed')) {
                bioContentText.setAttribute('tabindex', '-1');
                bioContentText.setAttribute('aria-hidden', 'true');
            }
            
            // Add keyboard event listener
            button.addEventListener('keydown', (event) => {
                if (event.key === 'Enter' || event.key === ' ') {
                    event.preventDefault();
                    button.click();
                }
            });

            // Add click event listener
            button.addEventListener('click', () => {
                const parent = button.parentElement;
                const bioContentText = parent.querySelector('.bio-content-text');
                let isOpen = button.classList.contains('open');
                
                if (isOpen) {
                    bioContentText.classList.add('collapsed');
                    button.classList.remove('open');
                    button.classList.add('closed');
                    button.textContent = 'Show more';
                    button.setAttribute('aria-expanded', 'false');
                    button.setAttribute('aria-label', 'Expand bio content');
                    // Make collapsed content not keyboard accessible
                    bioContentText.setAttribute('tabindex', '-1');
                    bioContentText.setAttribute('aria-hidden', 'true');
                } else {
                    bioContentText.classList.remove('collapsed');
                    button.classList.remove('closed');
                    button.classList.add('open');
                    button.textContent = 'Show less';
                    button.setAttribute('aria-expanded', 'true');
                    button.setAttribute('aria-label', 'Collapse bio content');
                    // Make expanded content keyboard accessible
                    bioContentText.removeAttribute('tabindex');
                    bioContentText.setAttribute('aria-hidden', 'false');
                }
            });
        });
    });
</script>
